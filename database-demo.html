💻 SQL-скрипты создания таблиц

Таблица users

CREATE TABLE users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(255) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    car_brand VARCHAR(100),
    car_model VARCHAR(100),
    car_year INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

Таблица services

CREATE TABLE services (
    id INT PRIMARY KEY AUTO_INCREMENT,
    service_name VARCHAR(255) NOT NULL,
    category ENUM('repair', 'tuning') NOT NULL,
    price DECIMAL(10,2) NOT NULL,
    description TEXT
);

Таблица service_history

CREATE TABLE service_history (
    id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    service_id INT NOT NULL,
    service_date DATE NOT NULL,
    price_paid DECIMAL(10,2) NOT NULL,
    status ENUM('pending', 'completed', 'cancelled') DEFAULT 'pending',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (service_id) REFERENCES services(id) ON DELETE CASCADE
);

Таблица reviews

CREATE TABLE reviews (
    id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT,
    author_name VARCHAR(255) NOT NULL,
    rating INT CHECK(rating >= 1 AND rating <= 5),
    review_text TEXT NOT NULL,
    car_info VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE SET NULL
);

Получение истории услуг пользователя

SELECT 
    u.name AS user_name,
    s.service_name,
    sh.service_date,
    sh.price_paid,
    sh.status
FROM service_history sh
JOIN users u ON sh.user_id = u.id
JOIN services s ON sh.service_id = s.id
WHERE u.id = 1
ORDER BY sh.service_date DESC;

Статистика по пользователю

SELECT 
    u.name,
    u.email,
    COUNT(sh.id) AS total_services,
    SUM(sh.price_paid) AS total_spent
FROM users u
LEFT JOIN service_history sh ON u.id = sh.user_id
WHERE u.id = 1
GROUP BY u.id;
